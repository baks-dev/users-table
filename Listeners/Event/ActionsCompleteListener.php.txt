declare(strict_types=1);

namespace BaksDev\Users\UsersTable\Listeners\Event;


use BaksDev\Users\UsersTable\Type\Actions\Complete\ActionsCompleteType;
use BaksDev\Users\UsersTable\Type\Actions\Complete\Collection\ActionsCompleteCollection;
use Symfony\Component\Console\ConsoleEvents;
use Symfony\Component\Console\Event\ConsoleCommandEvent;
use Symfony\Component\EventDispatcher\Attribute\AsEventListener;
use Symfony\Component\HttpKernel\Event\ControllerEvent;

/**
 * Слушатель инициирует ActionsComplete для определения в типе доктрины.
 */
#[AsEventListener(event: ControllerEvent::class)]
#[AsEventListener(event: ConsoleEvents::COMMAND)]
final class ActionsCompleteListener
{

    private ActionsCompleteCollection $collection;

    public function __construct(ActionsCompleteCollection $collection)
    {
        $this->collection = $collection;
    }


    public function onKernelController(ControllerEvent $event): void
    {
        // Инициируем цвета
        if(in_array(ActionsCompleteType::class, get_declared_classes(), true))
        {
            $this->collection->cases();
        }
    }
    
    public function onConsoleCommand(ConsoleCommandEvent $event): void
    {
        // Всегда инициируем цвет в консольной комманде
        $this->collection->cases();
    }

}
